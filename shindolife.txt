function GetHttp(URL)
	local returning = nil
    repeat
        pcall(function()
            returning = game:HttpGet(URL,true)
        end)
        if returning == nil then
            warn("Failed to get "..URL)
            wait(0.5)
        end
    until returning ~= nil
	return returning
end
